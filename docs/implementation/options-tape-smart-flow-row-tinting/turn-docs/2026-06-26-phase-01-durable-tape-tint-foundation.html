Options Tape Smart-Flow Tinting Phase 01

Added the shared DurableTape row hook foundation and pure OptionsTape tint helpers for smart-flow metadata, without changing live subscriptions, API payloads, route composition, historical hydration, or durable option row view models.

Outcome

DurableTapeProps now exposes getRowClassName and getRowStyle hooks with typed row inputs containing item, rowKey, and index. Hook resolution is gated by the existing rowTinting feature flag, and the inline style merge keeps the virtual row height, transform, and grid template authoritative.

apps/web/features/options-tape/tinting.ts adds pure helpers that map smart-flow hypothesis type, direction, policy confidence, evidence quality, and abstention reasons into row class/style/metadata. Existing decorByTraceId entries now use the DurableTape row hooks, so the default feature pack keeps its behavior except that rowTinting now controls visible row decoration.

+export const getOptionsTapeSmartFlowRowTint = (
+  projection: OptionsTapeSmartFlowTintInput
+): OptionsTapeRowTint => {
+  const hypothesisType = projection.hypothesis.hypothesis_type;
+  const abstained = projection.abstention.abstained;
+  const direction = abstained
+    ? "abstained"
+    : normalizeTintDirection(projection.hypothesis.direction);
+  const policyConfidence = roundUnit(
+    projection.hypothesis.scores.confidence.policy_confidence
+  );
+  const evidenceQuality = roundUnit(
+    projection.evidence?.evidence_quality ??
+      projection.hypothesis.scores.confidence.evidence_quality
+  );
+  return {
+    className: [
+      "options-tape-row-tinted",
+      "options-tape-smart-flow-row",
+      `options-tape-row-hypothesis-${normalizeClassToken(hypothesisType)}`,
+      `options-tape-row-direction-${direction}`,
+      abstained ? "options-tape-row-abstained" : "",
+      `classifier-${TONE_BY_DIRECTION[direction]}`
+    ].filter(Boolean).join(" "),
+    style: tintStyle(intensity),
+    metadata: { source: "smart-flow", family: hypothesisType, direction, abstained }
+  };
+};
