Implementation Notes

  • Composed the dashboard from domain modules instead of terminal-specific tape panes: OptionsTape, FlowPacketsTape, EquitiesTape, and the existing AlertsModule now occupy explicit route grid areas.
  • Kept /options tape-first and connected option row packet focus to the durable flow packet focus callback. /tape continues to redirect to /options, and /news continues to use the extracted NewsWire module.
  • Added createStaticEquitiesTapeSource so route/dashboard composition can feed compact equities templates through the durable source contract without reaching into terminal pane state.
  • Retired the legacy terminal-owned OptionsPane, FlowPane, NewsPane, and news control rails after imports proved they were no longer reachable from production routes.
  • Removed stale table sizing CSS that forced horizontal scrolling in the old terminal panes, and gave the production /options tape a bounded height so virtual scrolling, scroll hold, and jump-to-live are available inside the tape.

diff --git a/apps/web/features/equities-tape/source.ts b/apps/web/features/equities-tape/source.ts

+++ b/apps/web/features/equities-tape/source.ts
@@
+const filterStaticEquityPrints = (
+  prints: readonly EquityPrint[],
+  scope: EquitiesTapeScope | undefined,
+  filters: DurableTapeFilterState
+) =>
+  prints.filter((print) => {
+    if (!matchesEquitiesTapeScope(print, scope)) {
+      return false;
+    }
+    if (!filters.includeOffExchange && print.offExchange) {
+      return false;
+    }
+    if (filters.venues.size > 0 && !filters.venues.has(print.venue)) {
+      return false;
+    }
+    return true;
+  });
+
+export const createStaticEquitiesTapeSource = (
+  prints: readonly EquityPrint[]
+): DurableTapeSource<EquityPrint, EquitiesTapeScope, DurableTapeFilterState> => ({
+  mode: "static",
+  subscribe: ({ scope, filters, onItems }) => {
+    onItems(filterStaticEquityPrints(prints, scope, filters), { reset: true });
+    return () => undefined;
+  },
+  loadOlder: async () => ({ items: [], exhausted: true })
+});
