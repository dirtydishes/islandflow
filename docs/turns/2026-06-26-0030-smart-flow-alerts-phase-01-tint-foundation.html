Smart-Flow Alerts Phase 01 Tint Foundation

Extracted reusable smart-flow tint semantics from Options Tape into a shared frontend module while preserving Options Tape row behavior, durable option row tinting, and the existing runtime contract surface.

Summary

Added apps/web/features/smart-flow as the shared home for smart-flow tint types, hypothesis tone mapping, direction normalization, confidence and evidence bands, abstention intensity, classifier intensity style generation, and summary labels.

Reworked apps/web/features/options-tape/tinting.ts so Options Tape keeps its evidence ref expansion, direct-print precedence, and options-tape-* class assembly while delegating generic smart-flow presentation policy to the shared module.

Reviewer hardening restored canonical @islandflow/types inputs and metadata for the shared tint API, preserving exhaustive hypothesis coverage as the contract evolves.

Changed Files

  • apps/web/features/smart-flow/tinting.ts: shared tint policy and contract-typed exported stable types.
  • apps/web/features/smart-flow/index.ts: feature-level exports for future consumers.
  • apps/web/features/smart-flow/tinting.test.ts: generic tint matrix tests backed by the canonical hypothesis schema.
  • apps/web/features/options-tape/tinting.ts: Options Tape wrapper over the shared policy.
  • apps/web/features/options-tape/options-tape.test.ts: narrowed Options Tape tests to row classes and option-specific mapping.

+  directional_accumulation: "green",

+export const getSmartFlowTint = (projection: SmartFlowTintInput): SmartFlowTint => {
+  const hypothesisType = projection.hypothesis.hypothesis_type;
+  const abstained = projection.abstention.abstained;
+  const direction = abstained
+    ? "abstained"
+    : normalizeSmartFlowDirection(projection.hypothesis.direction);
+  const policyConfidence = roundSmartFlowUnit(
+    projection.hypothesis.scores.confidence.policy_confidence
+  );
+  const evidenceQuality = roundSmartFlowUnit(projection.evidence.evidence_quality);
+  const confidenceBand = getSmartFlowPolicyConfidenceBand(policyConfidence);
+  const evidenceQualityBand = getSmartFlowEvidenceQualityBand(evidenceQuality);
+  const tone = abstained ? "neutral" : getSmartFlowHypothesisTone(hypothesisType);
+  const intensity = getSmartFlowTintIntensity({
+    abstained,
+    evidenceQuality,
+    policyConfidence
+  });
+
+  return {
+    style: getSmartFlowTintStyle(intensity),
+    metadata: {
+      family: hypothesisType,
+      hypothesisType,
+      tone,
+      intensity,
+      direction,
+      policyConfidence,
+      confidenceBand,
+      evidenceQuality,
+      evidenceQualityBand,
+      abstained,
+      abstentionReasons: getAbstentionReasons(projection.abstention),
+      sourceReasons: [...projection.abstention.source_reasons]
+    }
+  };
+};
