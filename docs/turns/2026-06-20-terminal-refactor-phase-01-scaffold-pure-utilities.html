Summary

Pure terminal helpers now live under apps/web/features/terminal/. The terminal route file imports and re-exports those helpers, preserving the public test-facing contract while leaving React state, DOM hooks, websocket code, and JSX rendering in place.

Changes Made

  • Added focused modules for config, routes, formatting, filters, tape/history helpers, evidence refs, chart markers, and shared types.
  • Moved route normalization, live manifest construction, virtual-list config, option filter helpers, ticker parsing, pausable tape reducers, pinned evidence pruning, evidence ref helpers, and chart marker selection out of terminal.tsx.
  • Kept terminal.tsx as the compatibility facade with the helper exports used by apps/web/app/terminal.test.ts.
  • Preserved route behavior for /, /options, /news, and /tape redirect compatibility.

+export const normalizeTerminalPathname = (pathname: string): string => {
+  if (pathname === "/tape") {
+    return "/options";
+  }
+  return KNOWN_TERMINAL_PATHS.has(pathname) ? pathname : "/";
+};
+
+export const getRouteFeatures = (pathname: string): RouteFeatures => {
+  const normalizedPath = normalizeTerminalPathname(pathname);
+  switch (normalizedPath) {
+    case "/options":
+      return { options: true, nbbo: true, flow: true, showOptionsPane: true, showFlowPane: true, needsClassifierDecor: true };
+    case "/news":
+      return { news: true, showNewsPane: true };
+    default:
+      return { options: true, equities: true, flow: true, news: true, alerts: true, smartMoney: true };
+  }
+};
